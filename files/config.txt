# Raspberry Pi Configuration

# Enable I2S audio
dtparam=i2s=on
dtoverlay=hifiberry-dac

# Bluetooth configuration
dtparam=krnbt=on
enable_uart=1
dtoverlay=disable-bt

# Audio settings
dtparam=audio=on
audio_pwm_mode=2

# Memory split for GPU (minimal for headless)
gpu_mem=16

# Enable SSH
enable_ssh=1

# HDMI settings (can be disabled for headless operation)
hdmi_force_hotplug=1
hdmi_group=1
hdmi_mode=1

# I2C and SPI (if needed for audio hardware)
dtparam=i2c_arm=on
dtparam=spi=on

# Performance settings
over_voltage=2
arm_freq=1000
core_freq=500
sdram_freq=500
force_turbo=0
